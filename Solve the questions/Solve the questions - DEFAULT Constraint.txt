--Task 1 – Students Table

--Create a table Students with columns:

--ID → primary key
--Name → not null
--Age → not null
--Country → default value = 'Algeria'


CREATE TABLE Students (
   ID int NOT NULL primary key,
   Name varchar(255) NOT NULL,
   Age int NOT NULL,
   Country  varchar(255) DEFAULT 'Algeria'
);



--Task 2 – Products Table
--Create a table Products with columns:
--ProductID → primary key
--ProductName → not null
--Price → not null
--Stock → default = 0
--CreatedAt → default = current date

CREATE TABLE Products (
   ID int NOT NULL primary key,
   ProductName varchar(255) NOT NULL,
   Price smallmoney NOT NULL,
   Stock int DEFAULT  0,
   CreatedAt  date DEFAULT GETDATE()
);

-- Task 5 – Accounts Table
--Create a table Accounts with columns:
--ID → primary key
--UserName → not null
--Status → default = 'Active'
--CreatedAt → default = current date


CREATE TABLE Accounts (
   ID int NOT NULL primary key,
   UserName varchar(255) NOT NULL,

);
